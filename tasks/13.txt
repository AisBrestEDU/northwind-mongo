Create a SQL query to count current and discontinued products:

| TotalOfCurrentProducts | TotalOfDiscontinuedProducts |

var numberOfProducts = db.products.count({});
db.products.aggregate([
    {$match: {Discontinued: 1}},
    {$group: {_id: "$Discontinued", TotalNumberOfProducts: {$sum: 1}}},
    {$project: {_id:0, "TotalOfCurrentProducts": {$avg : numberOfProducts}, "TotalOfDiscontinuedProducts": "$TotalNumberOfProducts"}}
]);
