Create a SQL query to count current and discontinued products:

| TotalOfCurrentProducts | TotalOfDiscontinuedProducts |

var discontinuedProductsNumber = db.products.count({Discontinued: 1});
var currentProductsNumber = db.products.count({Discontinued: 0});
db.products.aggregate([
    {$project: {_id:0, TotalOfCurrentProducts: {$literal: currentProductsNumber}, TotalOfDiscontinuedProducts: {$literal: discontinuedProductsNumber}}},
    {$limit: 1}
]);
