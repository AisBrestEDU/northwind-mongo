Create SQL query to display the average price of each categories's products:

| CategoryName | AvgPrice |



db.products.aggregate([
    {
        $group: 
        { 
            _id: "$CategoryID",
            "Avg": { $avg: "$UnitPrice" }
        }
    },
    {
        $lookup:
        {
            from: "categories",
            localField: "_id",
            foreignField: "CategoryID",
            as: "category"
        }
    },
    {
        $project:
        {
            _id:0, 
            "CategoryName": { $arrayElemAt: [ "$category.CategoryName", 0 ] },
            AvgPrice: "$Avg"
        }
    }
])
