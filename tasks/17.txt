Create SQL query to display the average price of each categories's products:

| CategoryName | AvgPrice |



db.categories.aggregate([
    {          
        $lookup:
        {
            from: "products",
            localField: "CategoryID",
            foreignField: "CategoryID",
            as: "products_obj"
        }        
    },
    {
        $project:
        {
            _id: 0,
            CategoryName:1, 
            AvgPrice: { $avg: "$products_obj.UnitPrice" }
        }
    }
])
