Create a SQL query to get Product list (name, unit price) where products cost between $5 and $15:

| ProductName | UnitPrice |

db.products.aggregate([
    {
        $match: { UnitPrice: { $gt: 5, $lt: 15 } }
    },
    {
        $project:
        {
            _id: 0,
            ProductName: 1,
            UnitPrice: 1
        }
    }
])
