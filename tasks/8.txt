Create a query to return:

| CategoryName | TotalNumberOfProducts |

db.categories.aggregate([
    {$lookup: 
        {
          from: "products",
          localField: "CategoryID",
          foreignField: "CategoryID",
          as: "Products"
        }},
        {$unwind: "$Products"},
        {$group: {_id: "$CategoryName", TotalNumberOfProducts: {$sum: 1}}},
        {$project: {_id: 0, CategoryName: "$_id", TotalNumberOfProducts: "$TotalNumberOfProducts"}}
    ]);
